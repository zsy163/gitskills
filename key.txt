#include <iostream>
#include<string>
using namespace std;
int main(){
	string s;
	int cnt=1;
	while(s.length()>16||cnt==1){
		cout<<"请输入不密码(每个字符为数字或者大小写字母，长度不超过16)：";
		getline(cin,s);
		for(int i=0;i<s.length();i++)
		{
			cnt=0;
			int k;
			k=s[i]-'0'; 
			if(k>=0&&k<=9||k>=17&&k<=42||k>=49&&k<=74)
			{
				continue;
			} 
			else
			{
				cnt=1;
				break;
			} 
		}
		if(s.length()<=16&&cnt==0)
		{
			break;
		}
	}
	int j=0;
	for(int i=0;i<s.length();i++)
	{
		j=i;
		s[i]=s[i]+(j+1)+3;
	}
	char temp;
	temp=s[0];
	s[0]=s[j];
	s[j]=temp;
	for(int i=j;i>=0;i--)
	{
		cout<<s[i];
	}
	return 0;
}
