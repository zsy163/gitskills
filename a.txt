#include <iostream>
#include<string>
using namespace std;
int main(){
	string s;
	int cnt=1,a=0,b=0,c=0;;
	while(s.length()>16||cnt==1){
		cout<<"请输入不密码(每个字符为数字或者大小写字母，长度不超过16)：";
		getline(cin,s);
		for(int i=0;i<s.length();i++)
		{
			cnt=0;
			int k;
			k=s[i]-'0'; 
			if(k>=0&&k<=9)
			{
				a++;
				continue;
			} 
			else if(k>=17&&k<=42) 
			{
				b++;
				continue;
			}
			else if(k>=49&&k<=74)
			{
				c++;
				continue;
			}
			else
			{
				cnt=1;
				break;
			} 
		}
		if(s.length()<=16&&cnt==0)
		{
			break;
		}
	}
	int j=0;
	for(int i=0;i<s.length();i++)
	{
		j=i;
		s[i]=s[i]+(j+1)+3;
		//将每个字符的ASCII码加上它在字符串中的位置和偏移值3
	}
	char temp;
	temp=s[0];
	s[0]=s[j];
	s[j]=temp;
	//将字符串的第一位和最后一位调换顺序
	for(int i=j;i>=0;i--)
	{
		cout<<s[i];
		//将字符串反转并输出
	}
	cout<<endl;
	cout<<"密码强度为：";
	if(s.length()>=8&&a>0&&b>0&&c>0)
	{
		cout<<"高强度"<<endl;
		exit(0);
	}
	if(s.length()>=8&&a>0&&b>0||s.length()>=8&&a>0&&c>0)
	{
		cout<<"中强度"<<endl;
	}
	if(s.length()<8&&a>0&&b==0&&c==0||s.length()<8&&b>0&&a==0||s.length()<8&&c>0&&a==0||s.length()>=8&&a>0&&b==0&&c==0||s.length()>=8&&b>0&&a==0||s.length()>=8&&c>0&&a==0)
	{
		cout<<"弱强度"<<endl;
	}
	return 0;
}
