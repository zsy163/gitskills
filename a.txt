//容错处理 
void wrong(char *s){
	int cnt=1; 	
	int k=0;
	k=strlen(s);
	while(strlen(s)>16||cnt==1){
		cout<<"请输入密码(每个字符为数字或者大小写字母，长度不超过16)：";
		cin>>s;
		for(int i=0;i<strlen(s);i++)
		{
			cnt=0;
			if(s[i] < '0' && s[i]>'9' && s[i] < 'a' && s[i]>'z' && s[i] < 'A' && s[i]>'Z')
			{
				cnt=1;
				break;
			} 
			else
			{
				continue;
			} 
		}
		if(strlen(s)<=16&&cnt==0)
		{
			break;
		}
	}
}
//加密功能 
void save(char* s) {
	wrong(s);
	int j=0;
	for(int i=0;i<strlen(s);i++)
	{
		j=i;
		s[i]=s[i]+(j+1)+3;
		//将每个字符的ASCII码加上它在字符串中的位置和偏移值3
	}
	char temp;
	temp=s[0];
	s[0]=s[j];
	s[j]=temp;
	//将字符串的第一位和最后一位调换顺序
	for(int i=j;i>=0;i--)
	{
		cout<<s[i];
		//将字符串反转并输出
	}
}
//解密功能
void open(char* s){
	wrong(s);
	// 将字符串反转
    for (int i = 0; i < strlen(s) / 2; i++) {
        char temp = s[i];
        s[i] = s[strlen(s) - 1 - i];
        s[strlen(s) - 1 - i] = temp;
    }
	// 将字符串的第一位和最后一位调换顺序
    char temp = s[0];
    s[0] = s[strlen(s) - 1];
    s[strlen(s) - 1] = temp;
    // 将每个字符的ASCII码减去它在字符串中的位置和偏移值
    for (int i = 0; i < strlen(s); i++) {
        s[i] = s[i] - (i + 1) - 3;
    }
    cout << "解密后的密码：" << s << endl;

}
//判断密码强度
void  think(char *s)
{
	int cnt=1,a=0,b=0,c=0;;
	while(strlen(s)>16||cnt==1){
		cout<<"请输入不密码(每个字符为数字或者大小写字母，长度不超过16)：";
		cin>>s;
		for(int i=0;i<strlen(s);i++)
		{
			cnt=0;
			int k;
			k=s[i]-'0'; 
			if(k>=0&&k<=9)
			{
				a++;
				continue;
			} 
			else if(k>=17&&k<=42) 
			{
				b++;
				continue;
			}
			else if(k>=49&&k<=74)
			{
				c++;
				continue;
			}
			else
			{
				cnt=1;
				break;
			} 
		}
		if(strlen(s)<=16&&cnt==0)
		{
			break;
		}
	}
	cout<<"密码强度为：";
	if (strlen(s) < 8) {
        cout << "弱强度" << endl;
    }
	else{ 
		if (a != 0 && c != 0 && b != 0) {
            cout << "高强度" << endl;
            exit(0);
        }
        if (a != 0 && (b + c) != 0) {
            cout << "中强度" << endl;
        }
        else {
            cout << "高强度" << endl;
        }

	}
}
//密码生成 
void creat(char* s) {
    int len, choice;
    cout << "请输入密码的长度:    " ;
    cin >> len;
    srand((unsigned int)time(NULL));
    s[0] = '0' + rand() % 10;
    s[1] = 'a' + rand() % 26;
    s[2] = 'A' + rand() % 26;
    for (int i = 3; i < len; i++) {
        int type = rand() % 3;
        if (type == 0) {
            s[i] = '0' + rand() % 10;
        }
        else if (type == 1) {
            s[i] = 'a' + rand() % 26;
        }
        else {
            s[i] = 'A' + rand() % 26;
        }
    }
    for (int i = 0; i < len; i++) {
        int j = rand() % len;
        char temp = s[i];
        s[i] = s[j];
        s[j] = temp;
    }
    cout << "生成的强密码为：";
    for(int i=0;i<len;i++)
    {
    	cout<<s[i];
	}
    cout <<endl;
}
